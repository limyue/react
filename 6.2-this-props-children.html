this.props.children

表示组件所有子节点（列表展示不同内容/个数不同项）
this.props.children 从父组件获取需要展示的列表内容
获取到列表项之后，需要遍历children，逐项设置。使用React.children.map
注意：有一个返回值是数组

React.Children.map(this.props.children,function(child){
	<li>{child}</li>
})



var List = React.createClass({
 	render: function(){
 	return(
		<ul>
			{	//React.Children.map()  大写Children
			React.Children.map(this.props.children,function(child){
				return <li>{child}</li>
			})
			}
		</ul>
 	)
 	}
 })

 ReactDOM.render(
	(<List>
		<h1>标题</h1>
		<a href="#">内容1</a>
		<a href="#">内容2</a>
	</List>),
	document.getElementById('container')
 )
