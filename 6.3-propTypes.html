属性验证 propTypes 
 
验证组件属性是否符合要求
React.PropTypes.string.isRequired 验证是否是字符串


 var ShowTitel = React.createClass({
 	propTypes: {
 		titel: React.PropTypes.string.isRequired
 	},
 	render: function(){
 		return <h1>{this.props.titel}</h1>
 	}
 })

 ReactDOM.render(
	<ShowTitel titel="123"/>,
	document.getElementById('container')
 )


设置属性默认值 
getDefaultProps: function(){
	return {
		titel: '默认titel'	//titel的默认值
	}
}



var ShowTitel = React.createClass({
 	getDefaultProps: function(){
 		return {
 			titel: '默认titel'
 		}
 	},
 	render: function(){
 		return <h1>{this.props.titel}</h1>
 	}
 })

 ReactDOM.render(
	<ShowTitel/>,
	document.getElementById('container')
 )
