//react处理事件
var MyButton = React.createClass({
 	handleClick:function(){
 		console.log('点击成功')
 	},
 	render:function(){
 		return <button onClick={this.handleClick}>{this.props.button}</button>
 	}
 })
  ReactDOM.render(
	<MyButton button="按钮"/>,
	document.getElementById('container')
  )
